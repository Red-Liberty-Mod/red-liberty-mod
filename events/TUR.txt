country_event = { # deal presented to angloids and frogs
	id = anatalia_conference.2
	title = anatalia_conference.2.title
	desc = anatalia_conference.2.desc
	
	fire_only_once = yes
	is_triggered_only = yes
	
	option = { # accept
		name = anatalia_conference.2.a
		FRA = {
			country_event = {
				id = anatalia_conference.3
				hours = 6
			}
		}
		
		ai_chance = {
			base = 10
			modifier = {
				factor = 0
				check_variable = { TUR.TUR_demand_val > TUR.TUR_concession_val }
			}
		}
	}
	
	option = { # reject
		name = anatalia_conference.2.b
		if = {
			limit = { check_variable = { TUR.TUR_concession_val > TUR.TUR_demand_val } }
			custom_effect_tooltip = TUR_rejecting_favorable_deal
			add_political_power = -100
		}
		
		TUR = {
			set_country_flag = TUR_british_rejected_deal
			add_opinion_modifier = {
				target = ENG
				modifier = TUR_rejected_deal
			}
			country_event = {
				id = anatalia_conference.5
				hours = 6
			}
		}
		
		ai_chance = {
			base = 10
			modifier = {
				factor = 0
				check_variable = {
					var = TUR.TUR_demand_val
					value = TUR.TUR_concession_val
					compare = less_than_or_equals
				}
			}
		}
	}
}

country_event = { # conference successful
	id = anatalia_conference.4
	title = anatalia_conference.4.title
	desc = anatalia_conference.4.desc
	
	fire_only_once = yes
	is_triggered_only = yes
	
	option = {
		name = anatalia_conference.4.a
		TUR = {
			annex_country = {
				target = SCM
				transfer_troops = no
			}
			if = {
				limit = {
					has_country_flag = TUR_demanded_manisa
				}
				339 = {
					set_demilitarized_zone = no
				}
			}
			if = {
				limit = {
					has_country_flag = TUR_renegotiated_debt
				}
				add_timed_idea = {
					idea = TUR_renegotiated_debt
					days = 360
				}
			}
			if = {
				limit = {
					has_country_flag = TUR_claimed_thrace
				}
				add_state_claim = 821
			}
			if = {
				limit = {
					has_country_flag = TUR_remil_lemnos
				}
				993 = {
					set_demilitarized_zone = no
				}
			}
			add_opinion_modifier = {
				target = ENG
				modifier = TUR_successful_conference
			}
			add_opinion_modifier = {
				target = FRA
				modifier = TUR_successful_conference
			}
		}
		hidden_effect = {
			set_global_flag = TUR_straits_conference_over
			SCM = {
				remove_state_core = 341
				remove_state_core = 345
			}
			news_event = {
				id = worldnews.14
				hours = 6
			}
		}
	}
}
