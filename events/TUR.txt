country_event = { # constructive negotiations
	id = anatalia_conference.20
	title = anatalia_conference.20.title
	desc = anatalia_conference.20.desc
	
	fire_only_once = yes
	is_triggered_only = yes
	
	option = { # press for full sovereignty
		name = anatalia_conference.20.a
		if = {
			limit = { check_variable = { TUR_nationalist_pressure_var < 70 } }
			country_event = {
				id = anatalia_conference.21
				hours = 6
			}
		}
		else = {
			country_event = {
				id = anatalia_conference.31
				hours = 6
			}
		}
		
		ai_chance = {
			base = 1
		}
	}
	
	option = { # accept the agreement as-is
		name = anatalia_conference.20.b
		country_event = {
			id = anatalia_conference.22
			hours = 6
		}
		
		ai_chance = {
			base = 9
		}
	}
}

		name = anatalia_conference.30.b
		add_to_variable = {
			tooltip = TUR_atmosphere_var_tt
			TUR_atmosphere_var = -5
		}
		hidden_effect = {
			if = {
				limit = { check_variable = { TUR_atmosphere_var < 46 } }
				country_event = {
					id = anatalia_conference.40
					hours = 6
				}
			}
			else = {
				country_event = {
					id = anatalia_conference.32
					hours = 6
				}
			}
		}
		
		ai_chance = {
			base = 4
			modifier = {
				factor = 2
				check_variable = { TUR_atmosphere_var > 50 }
			}
		}
	}
}
