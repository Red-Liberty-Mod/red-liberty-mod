SOV_find_chairman_options = {

	add_to_temp_array = {
		array = chairman_options
		value = SOV_industrialist_influence
	}
	add_to_temp_array = {
		array = chairman_options
		value = SOV_rightist_influence
	}
	add_to_temp_array = {
		array = chairman_options
		value = SOV_decist_influence
	}
	add_to_temp_array = {
		array = chairman_options
		value = SOV_syndicalist_influence
	}
	add_to_temp_array = {
		array = chairman_options
		value = SOV_democrat_influence
	}
	
	# find the faction with the highest influence
	find_highest_in_array = {
		array = chairman_options 
		index = chairman_options_index # (default 'i') index of highest value in array will be stored in this temp variable
	}
	if = {
		limit = {
			check_variable = { chairman_options_index = 0 }
		}
		set_country_flag = SOV_industrialist_chairman_option
		remove_from_temp_array = {
			array = chairman_options
			index = 0
		}
		add_to_temp_array = {
			array = chairman_options
			index = 0
			value = 1
		}
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index = 1 }
		}
		set_country_flag = SOV_rightist_chairman_option
		remove_from_temp_array = {
			array = chairman_options
			index = 1
		}
		add_to_temp_array = {
			array = chairman_options
			index = 1
			value = 1
		}
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index = 2 }
		}
		set_country_flag = SOV_decist_chairman_option
		remove_from_temp_array = {
			array = chairman_options
			index = 2
		}
		add_to_temp_array = {
			array = chairman_options
			index = 2
			value = 1
		}
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index = 3 }
		}
		set_country_flag = SOV_syndicalist_chairman_option
		remove_from_temp_array = {
			array = chairman_options
			index = 3
		}
		add_to_temp_array = {
			array = chairman_options
			index = 3
			value = 1
		}
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index = 4 }
		}
		set_country_flag = SOV_democrat_chairman_option
		remove_from_temp_array = {
			array = chairman_options
			index = 4
		}
		add_to_temp_array = {
			array = chairman_options
			index = 4
			value = 1
		}
	}
	
	# run the effect again, finding the second strongest faction
	find_highest_in_array = {
		array = chairman_options
		index = chairman_options_index_2 # (default 'i') index of highest value in array will be stored in this temp variable
	}
	#log = "chairman_options_index_2 = [?chairman_options_index_2]"
	if = {
		limit = {
			check_variable = { chairman_options_index_2 = 0 }
		}
		set_country_flag = SOV_industrialist_chairman_option
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index_2 = 1 }
		}
		set_country_flag = SOV_rightist_chairman_option
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index_2 = 2 }
		}
		set_country_flag = SOV_decist_chairman_option
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index_2 = 3 }
		}
		set_country_flag = SOV_syndicalist_chairman_option
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index_2 = 4 }
		}
		set_country_flag = SOV_democrat_chairman_option
	}
}

SOV_damage_relocated_military_factories_effect = {
	if = {
		limit = {
			FROM = {
				NOT = { is_fully_controlled_by = ROOT }
			}
		}
		damage_building = { type = arms_factory damage = 0.9 }
	}
	else = {
		damage_building = { type = arms_factory damage = 0.5 }
	}

SOV_calc_faction_effects = {
	
	### Centralists ###
	if = {
		limit = {
			check_variable = { SOV_industrialist_opinion < 15 }
		}
		set_variable = { SOV_centralist_stability_factor = -0.1 }
	}
	else = {
		set_variable = { SOV_centralist_stability_factor = 0.1 }
	}
	
	set_temp_variable = { SOV_centralist_effect_factor = SOV_industrialist_influence }
	divide_temp_variable = { SOV_centralist_effect_factor = 30 }
	multiply_variable = { SOV_centralist_stability_factor = SOV_centralist_effect_factor }
	
	### Rightists ###
	if = {
		limit = {
			check_variable = { SOV_rightist_opinion < 15 }
		}
		set_variable = { SOV_rightist_political_power_factor = -0.1 }
	}
	else = {
		set_variable = { SOV_rightist_political_power_factor = 0.1 }
	}
	
	set_temp_variable = { SOV_rightist_effect_factor = SOV_rightist_influence }
	divide_temp_variable = { SOV_rightist_effect_factor = 30 }
	multiply_variable = { SOV_rightist_political_power_factor = SOV_rightist_effect_factor }
	
	### Syndicalist ###
	if = {
		limit = {
			check_variable = { SOV_syndicalist_opinion < 15 }
		}
