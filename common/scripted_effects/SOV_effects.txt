
SOV_reduce_turmoil = {
	if = {
		limit = {
			has_idea = SOV_political_turmoil_1
		}
		swap_ideas = {
			remove_idea = SOV_political_turmoil_1
			add_idea = SOV_political_turmoil_2
		}
	}
	else_if = {
		limit = {
			has_idea = SOV_political_turmoil_2
		}
		swap_ideas = {
			remove_idea = SOV_political_turmoil_2
			add_idea = SOV_political_turmoil_3
		}
	}
	else_if = {
		limit = {
			has_idea = SOV_political_turmoil_3
		}
		swap_ideas = {
			remove_idea = SOV_political_turmoil_3
			add_idea = SOV_political_turmoil_4
		}
	}
	else_if = {
		limit = {
			has_idea = SOV_political_turmoil_4
		}
		swap_ideas = {
			remove_idea = SOV_political_turmoil_4
			add_idea = SOV_political_turmoil_5

	}
	else_if = {
		limit = {
			has_idea = SOV_political_turmoil_4
		}
		remove_ideas = SOV_political_turmoil_5
	}
}

SOV_find_chairman_options = {

	add_to_temp_array = {
		array = chairman_options
		value = SOV_industrialist_influence
	}
	add_to_temp_array = {
		array = chairman_options
		value = SOV_rightist_influence
	}
	add_to_temp_array = {
		array = chairman_options
		value = SOV_decist_influence
	}
	add_to_temp_array = {
		array = chairman_options
		value = SOV_syndicalist_influence
	}
	add_to_temp_array = {
		array = chairman_options
		value = SOV_democrat_influence
	}
	
	# find the faction with the highest influence
	find_highest_in_array = {
		array = chairman_options 
		index = chairman_options_index # (default 'i') index of highest value in array will be stored in this temp variable
	}
	if = {
		limit = {
			check_variable = { chairman_options_index = 0 }
		}
		set_country_flag = SOV_industrialist_chairman_option
		remove_from_temp_array = {
			array = chairman_options
			index = 0
		}
		add_to_temp_array = {
			array = chairman_options
			index = 0
			value = 1
		}
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index = 1 }
		}
		set_country_flag = SOV_rightist_chairman_option
		remove_from_temp_array = {
			array = chairman_options
			index = 1
		}
		add_to_temp_array = {
			array = chairman_options
			index = 1
			value = 1
		}
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index = 2 }
		}
		set_country_flag = SOV_decist_chairman_option
		remove_from_temp_array = {
			array = chairman_options
			index = 2
		}
		add_to_temp_array = {
			array = chairman_options
			index = 2
			value = 1
		}
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index = 3 }
		}
		set_country_flag = SOV_syndicalist_chairman_option
		remove_from_temp_array = {
			array = chairman_options
			index = 3
		}
		add_to_temp_array = {
			array = chairman_options
			index = 3
			value = 1
		}
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index = 4 }
		}
		set_country_flag = SOV_democrat_chairman_option
		remove_from_temp_array = {
			array = chairman_options
			index = 4
		}
		add_to_temp_array = {
			array = chairman_options
			index = 4
			value = 1
		}
	}
	
	# run the effect again, finding the second strongest faction
	find_highest_in_array = {
		array = chairman_options
		index = chairman_options_index_2 # (default 'i') index of highest value in array will be stored in this temp variable
	}
	#log = "chairman_options_index_2 = [?chairman_options_index_2]"
	if = {
		limit = {
			check_variable = { chairman_options_index_2 = 0 }
		}
		set_country_flag = SOV_industrialist_chairman_option
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index_2 = 1 }
		}
		set_country_flag = SOV_rightist_chairman_option
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index_2 = 2 }
		}
		set_country_flag = SOV_decist_chairman_option
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index_2 = 3 }
		}
		set_country_flag = SOV_syndicalist_chairman_option
	}
	else_if = {
		limit = {
			check_variable = { chairman_options_index_2 = 4 }
		}
		set_country_flag = SOV_democrat_chairman_option
	}
}

SOV_init_evacuation_vars = {
	set_variable = { SOV_industrial_relocation_max_factories = 4 }
	set_variable = { SOV_industrial_relocation_trains_decision = 1 }
	set_variable = { SOV_industrial_relocation_trains_decision_negative = -1 }
}

SOV_damage_relocated_military_factories_effect = {
	if = {
		limit = {
			FROM = {
				NOT = { is_fully_controlled_by = ROOT }
			}
		}
		damage_building = { type = arms_factory damage = 0.9 }
	}
	else = {
		damage_building = { type = arms_factory damage = 0.5 }
	}
}

			}
		}

	}

		damage_building = { type = industrial_complex damage = 0.5 }
	}
}

	if = {
		limit = {
			OR = {
				has_state_flag = SOV_civs_removed_flag

		}
		ROOT = {
			if = { #Prio the Urals
				limit = {
					any_controlled_state = {
						SOV_is_industrial_relocation_destiny_state = yes

							size > 1

						}
					}

							FROM = { has_state_flag = SOV_mils_removed_flag }
						}

							type = arms_factory
							level = 1
							instant_build = yes
						}
