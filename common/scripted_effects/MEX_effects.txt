MEX_setup_CRT = {
	hidden_effect = {
		MEX = {
			every_owned_state = {
				if = {
					limit = {
						has_dynamic_modifier = { modifier = MEX_cristero_revolt_heavy }
					}
					CRT = {
						transfer_state = PREV
					}
					set_demilitarized_zone = no
					remove_dynamic_modifier = {
						modifier = MEX_cristero_revolt_heavy
					}
				}
				else_if = {
					limit = {
						has_dynamic_modifier = { modifier = MEX_cristero_revolt_medium }
					}
					CRT = {
						set_state_controller = PREV
					}
					set_demilitarized_zone = no
					remove_dynamic_modifier = {
						modifier = MEX_cristero_revolt_medium
					}
				}
				else_if = {
					limit = {
						has_dynamic_modifier = { modifier = MEX_cristero_revolt_light }
					}
					set_demilitarized_zone = no
					remove_dynamic_modifier = {
						modifier = MEX_cristero_revolt_light
					}
				}
			}
		}
		CRT = {
			load_oob = "CRT"
			
			if = {
				limit = { has_template = "Brigada Agrarista" }
				delete_unit_template_and_units = { division_template = "Brigada Agrarista" }
			}
		}
	}
}
