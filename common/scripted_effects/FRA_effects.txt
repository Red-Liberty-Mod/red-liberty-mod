FRA_improve_govt_stability = {
	
	if = {
		limit = {
			has_idea = FRA_fallen_govt
		}
		swap_ideas = {
			remove_idea = FRA_fallen_govt
			add_idea = FRA_unstable_govt
		}
	}
	else_if = {
		limit = {
			has_idea = FRA_unstable_govt
		}
		remove_ideas = FRA_unstable_govt
	}
}
