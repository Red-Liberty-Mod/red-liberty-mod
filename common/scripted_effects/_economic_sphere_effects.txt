@MEMBER_STATUS_LEADER = 1
@MEMBER_STATUS_MEMBER = 2
@MEMBER_STATUS_OBSERVER = 3

create_mitteleuropa_sphere = {
	GER = {
		add_THIS_to_mitteleuropa_sphere = yes
		set_variable = { econ_sphere_member_status_var = @MEMBER_STATUS_LEADER }
	}
	AHE = { add_THIS_to_mitteleuropa_sphere = yes }
	
	FIN = { add_THIS_to_mitteleuropa_sphere = yes }
	BUL = {
		add_THIS_to_mitteleuropa_sphere = yes
		set_variable = { econ_sphere_member_status_var = @MEMBER_STATUS_OBSERVER }
	}
	BAL = { add_THIS_to_mitteleuropa_sphere = yes }
	POL = { add_THIS_to_mitteleuropa_sphere = yes }
	LIT = { add_THIS_to_mitteleuropa_sphere = yes }
	BLR = { add_THIS_to_mitteleuropa_sphere = yes }
	UKR = { add_THIS_to_mitteleuropa_sphere = yes }
	GEO = { add_THIS_to_mitteleuropa_sphere = yes }
}
add_THIS_to_mitteleuropa_sphere = {
	custom_effect_tooltip = add_to_mitteleuropa_tt
	add_to_array = { global.mitteleuropa_members = THIS }
	set_variable = { econ_sphere_member_status_var = 2 }
	log = "[GetDateText]: [THIS.GetName]: effect add_THIS_to_mitteleuropa_sphere"
	
	for_each_scope_loop = {
		array = global.mitteleuropa_members
		if = {
			limit = {
				NOT = { has_market_access_with = PREV }
				NOT = { tag = PREV }
			}
			diplomatic_relation = {
				country = PREV
				relation = market_access_rights
			}
		}
	}
}
remove_THIS_from_mitteleuropa_sphere = {
	custom_effect_tooltip = remove_from_mitteleuropa_tt
	remove_from_array = { global.mitteleuropa_members = THIS }
	clear_variable = econ_sphere_member_status_var
	
	for_each_scope_loop = {
		array = global.mitteleuropa_members
		if = {
			limit = { has_market_access_with = PREV }
			diplomatic_relation = {
				country = PREV
				relation = market_access_rights
				active = no
			}
		}
	}
}
