USA_setup_SPA_pie_chart_values = {
	
	set_variable = { SPA_unionist_pop = 27 }
	set_variable = { SPA_militant_pop = 10 }
	set_variable = { SPA_moderate_pop = 19 }
	
	set_variable = { USA_SPA_unity = 17 }
	
	create_pie_chart = yes
}

# this creates a pie chart of popularities - can be easily adapted to show anything that adds up to 100%
create_pie_chart = {
	clear_array = pie_chart
	resize_array = {
		array = pie_chart
		value = 1 # 1 is value for neutrality frame, so we don't have to make a loop for it
		size = 100 # 100 pieces, each percent is one piece
	}
	
	set_temp_variable = { demsoc_pop = SPA_unionist_pop }
	set_temp_variable = { revsoc_pop = SPA_militant_pop }
	set_temp_variable = { socdem_pop = SPA_moderate_pop }
	
	set_temp_variable = { spa_total_pop = demsoc_pop }
	add_to_temp_variable = { spa_total_pop = revsoc_pop }
	add_to_temp_variable = { spa_total_pop = socdem_pop }
	
	#log = "[This.GetName] spa_total_pop = [?USA.spa_total_pop]"
	
	divide_temp_variable = { demsoc_pop = spa_total_pop }
	divide_temp_variable = { revsoc_pop = spa_total_pop }
	divide_temp_variable = { socdem_pop = spa_total_pop }
	
	### LIBSOC ###
	#set_temp_variable = { demsoc_pop = party_popularity@libertarian_socialism }
	multiply_temp_variable = { demsoc_pop = 100 }
	round_temp_variable = demsoc_pop
	
	#log = "[This.GetName] demsoc_pop = [?USA.demsoc_pop]"

	### PROLEDICT ###
	#set_temp_variable = { revsoc_pop = party_popularity@proletarian_dictatorship }
	multiply_temp_variable = { revsoc_pop = 100 }
	round_temp_variable = revsoc_pop 
	set_variable = { ROOT.last_idx = demsoc_pop }
	add_to_temp_variable = { revsoc_pop = ROOT.last_idx }
	for_loop_effect = {
		start = ROOT.last_idx
		end = revsoc_pop
		set_variable = { pie_chart^v = 2 }
		add_to_variable = { ROOT.last_idx = 1 }
	}
	
	#log = "[This.GetName] revsoc_pop = [?USA.revsoc_pop]"

	### SOCDEM ###
	#set_temp_variable = { socdem_pop = party_popularity@social_democracy }
	multiply_temp_variable = { socdem_pop = 100 }
	round_temp_variable = socdem_pop
	add_to_temp_variable = { socdem_pop = ROOT.last_idx }
	for_loop_effect = {
		start = ROOT.last_idx
		end = socdem_pop
		set_variable = { pie_chart^v = 3 }
		add_to_variable = { ROOT.last_idx = 1 }
	}
	
	#log = "[This.GetName] socdem_pop = [?USA.socdem_pop]"

	### END ###
	clear_variable = ROOT.last_idx
	add_to_variable = { pie_chart_dirty = 1 }
}
