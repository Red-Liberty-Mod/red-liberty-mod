defined_text = {
	name = GetPresidentPopularity
	
	text = {
		trigger = {
			check_variable = { USA_president_popularity > 16 }
		}
		localization_key = "USA_broadly_popular"
	}
	text = {
		trigger = {
			check_variable = { USA_president_popularity > 12 }
			check_variable = { USA_president_popularity < 17 }
		}
		localization_key = "USA_divides_opinions"
	}
	text = {
		trigger = {
			check_variable = { USA_president_popularity > 6 }
			check_variable = { USA_president_popularity < 11 }
		}
		localization_key = "USA_unpopular"
	}
	text = {
		trigger = {
			check_variable = { USA_president_popularity < 7 }
		}
		localization_key = "USA_deeply_unpopular"
	}
}
