line_infantry = {
	blocked_for = {
		FRA USA GER
	}

	role = infantry

	upgrade_prio = {
		factor = 10
	}

	infantry_default = {
		upgrade_prio = {
			base = 1
		}

		target_template = {
			regiments = {
				infantry = 9
				artillery_brigade = 1
			}

			support = {
				artillery = 1
				anti_air = 1
				engineer = 1
				logistics_company = 1
				field_hospital = 1
			}
		}
	}

	infantry_motorised = {

		upgrade_prio = {
			base = 1
			modifier = {
				factor = 0
				NOT = { has_tech = motorised_infantry }
			}
			modifier = {
				factor = 0
				num_of_military_factories < 50
			}
			modifier = {
				add = 2
				num_of_military_factories > 99
			}
		}

		target_template = {
			regiments = {
				motorized = 9
				mot_artillery_brigade = 3
			}

			support = {
				engineer = 1
				artillery = 1
				anti_air = 1
				logistics_company = 1
				field_hospital = 1
			}
		}
	}

	infantry_mechanised = {

		upgrade_prio = {
			base = 1
			modifier = {
				factor = 0
				NOT = { has_tech = mechanised_infantry }
			}
			modifier = {
				factor = 0
				num_of_military_factories < 75
			}
			modifier = {
				add = 3
				num_of_military_factories > 149
			}
		}

		target_template = {
			regiments = {
				mechanized = 9
				mot_artillery_brigade = 3
			}

			support = {
				artillery = 1
				anti_air = 1
				engineer = 1
				logistics_company = 1
				field_hospital = 1
			}
		}
	}
	
	infantry_irregular = {
		can_upgrade_in_field = { always = yes }

			base = 0 #never, only here so the AI recognises irregulars as part of the infantry line
		}

		target_template = {
			regiments = {
				irregular_infantry = 5
			}
		}
	}

	infantry_militia = {
		can_upgrade_in_field = { always = yes }

		upgrade_prio = {
			base = 0 #never, only here so the AI recognises militias as part of the infantry line
		}

		target_template = {
			regiments = {
				militia = 5
			}
		}
	}
