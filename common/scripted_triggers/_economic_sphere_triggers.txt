is_in_economic_sphere = {
	custom_trigger_tooltip = {
		tooltip = is_in_economic_sphere_tt
		OR = {
			is_in_array = { global.mitteleuropa_members = FROM.id }
			is_in_array = { global.latin_union_members = FROM.id }
			is_in_array = { global.organization_of_american_states_members = FROM.id }
		}
	}
}

is_economic_sphere_leader = {
	custom_override_tooltip = {
		tooltip = is_economic_sphere_leader_tt
		is_in_economic_sphere = yes
		check_variable = { econ_sphere_member_status_var = 1 }
	}
}

is_in_same_economic_sphere_as_ROOT = {
	custom_override_tooltip = {
		tooltip = {
			localization_key = is_in_same_economic_sphere_as_tt
			COUNTRY = "[ROOT.GetNameWithFlag]"
		}
		is_in_economic_sphere = yes
		OR = {
			AND = {
				is_in_array = { global.mitteleuropa_members = ROOT.id }
				is_in_array = { global.mitteleuropa_members = THIS.id }
			}
			AND = {
				is_in_array = { global.latin_union_members = ROOT.id }
				is_in_array = { global.latin_union_members = THIS.id }
			}
			AND = {
				is_in_array = { global.organization_of_american_states_members = ROOT.id }
				is_in_array = { global.organization_of_american_states_members = THIS.id }
			}
			AND = {
				is_in_array = { global.comintern_members = ROOT.id }
				is_in_array = { global.comintern_members = THIS.id }
			}
		}
	}
}
