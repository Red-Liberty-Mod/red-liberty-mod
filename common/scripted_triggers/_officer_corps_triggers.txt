has_military_academy = {
	custom_trigger_tooltip = {
		tooltip = has_military_academy_tt
		if = {
			limit = {
				tag = LIB
			}
			has_completed_focus = LIB_officer_candidate_school
		}
		else = {
			always = yes
		}
	}
}
