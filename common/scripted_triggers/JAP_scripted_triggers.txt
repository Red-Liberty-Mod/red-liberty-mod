JAP_is_reformist_path = {
	custom_trigger_tooltip = {
		tooltip = JAP_is_reformist_path
		OR = {
			has_government = democratic_socialism
			has_government = social_democracy
			has_government = progressive_liberalism
		}
	}
}

JAP_is_conservative_path = {
	custom_trigger_tooltip = {
		tooltip = JAP_is_conservative_path
		OR = {
			has_government = liberal_conservatism
			has_government = national_conservatism
		}
	}
}

JAP_is_nationalist_path = {
	custom_trigger_tooltip = {
		tooltip = JAP_is_nationalist_path
		OR = {
			has_government = authoritarian_democracy
			has_government = despotism
			has_government = fascism
		}
	}
}
