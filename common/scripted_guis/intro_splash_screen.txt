scripted_gui = {
	
	intro_splash_screen_scripted_gui = {
		window_name = "intro_splash_screen_gui"
		context_type = player_context
		
		visible = {
			has_country_flag = open_intro_splash_screen
		}
		
		dynamic_lists = {
			entry_gridbox = {
				array = country_info_array
				entry_container = [GetCountryInfoEntryContainer]
				value = f
				index = g
			}
		}
		
		properties = {
			poster = {
				image = "[GetPoster]"
			}
		}
		
		effects = {
			content_button_click = {
				set_variable = { country_info_selected = f }
			}
			folder_button_click = {
				### EXPANDS ###
				if = {
					limit = {
						set_temp_variable = { check = g }
						add_to_temp_variable = { check = 1 }
						OR = {
							check_variable = { country_info_array^check > 100 }
							check_variable = { country_info_array^check = 0 }
						}
					}
					
					## Removes entries after the folder ##
					for_each_loop = {
						array = country_info_array
						if = {
							limit = {
								check_variable = { i > g }
							}
							add_to_array = { removed = v }
						}
					}
					for_each_loop = {
						array = removed
						if = {
							limit = {
								is_in_array = { array = country_info_array value = v }
							}
							remove_from_array = { array = country_info_array value = v }
						}
					}
					
					meta_effect = {
						text = {
							for_each_loop = {
								array = country_info_folder_array_[FOLDER]
								add_to_array = { country_info_array = v }
							}
						}
						FOLDER = "[?f]"
					}
					
					## Readds previously removed entries ##
					for_each_loop = {
						array = removed
						add_to_array = { country_info_array = v }
					}
					clear_array = removed
				}
				### COLLAPSES ###
				else = {
					meta_effect = {
						text = {
							for_each_loop = {
								array = country_info_folder_array_[FOLDER]
								if = {
									limit = {
										is_in_array = { array = country_info_array value = v }
									}
									remove_from_array = { array = country_info_array value = v }
								}
							}
						}
						FOLDER = "[?f]"
					}
				}
			}
			
			button_credits_click = {
				set_variable = { country_info_selected = 0 }
			}
			button_close_click = {
				clr_country_flag = open_intro_splash_screen
			}
		}
	}
}
