
FRA_focus_on_germany = {
	allowed = {
		original_tag = FRA
	}
	enable = {
		is_neighbor_of = GER
		GER = {
			is_faction_leader = yes
			has_capitulated = no
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = conquer
		id = GER
		value = 200
	}
	ai_strategy = {
		type = front_armor_score
		id = GER
		value = 100
	}
}

FRA_defend_belgian_border = {
	allowed = {
		original_tag = FRA
	}
	enable = {
		OR = {
			is_in_faction_with = BEL
			has_military_access_to = BEL
		}
	}
	abort_when_not_enabled = yes
	
	ai_strategy = {
		type = force_defend_ally_borders
		id = BEL
		value = 50
	}
}

FRA_endsieg = { # if losing, focus everything on defence
	
	allowed = {
		original_tag = FRA
	}
	enable = {
		surrender_progress > 0.25
		stockpile_ratio = { archetype = infantry_equipment ratio < 0.1 }
		GER = {
			has_war_with = ROOT
			has_capitulated = no
			surrender_progress < 0.1
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_control
		tag = GER
		priority = 150
		ordertype = front
		execute_order = no
	}
	ai_strategy = {
		type = invasion_unit_request
		area = globally
		value = -500
	}
	ai_strategy = {
		type = dont_defend_ally_borders
		id = BEL
		value = 100
	}
}
