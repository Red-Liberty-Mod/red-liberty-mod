		is_dynamic_country = no
		NOT = {
			OR = {
				has_country_leader_ideology = anarchism_subtype
				has_country_leader_ideology = anarcho_socialism_subtype
			}
		}
	}
	
	generate_character = {
		token_base = generic_anarchist_leader # token will be TAG_token_base

			ideology = anarchism_subtype
		}
		
		portraits = { 
			civilian = {
				large = "gfx/leaders/generic/Portrait_Generic_Anarchist_Commune.dds"
			}
		}
	}
}
